# Mercenary Enclave Greetings: Gatekeeper event
# Via on_custom_diplomacy | this = target country (player); from = country being contacted
country_event = {
	id = nations.11200
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		from = {
			is_country_type = PrimPlayer_ethic_fanatic_nation
			has_country_flag = ethic_fanatic_authoritarian_nation@root
		}
		NOT = {
			has_active_event = {
				nations.11205 #GREETINGS
				nations.11210 #HIRE FLEET
				nations.11211 #THANKS FOR RENTAL
				nations.11215 #MANAGE HIRED FLEET
				nations.11220 #MERCENARY GARRISON starbase building
				nations.11235 #Expand Mercenary Fleet
				nations.11240 #Tactical Appraisal
				nations.11245 #Tactical Appraisal Khan
				nations.11250 #Tactical Appraisal Gray Tempest
				nations.11255 #Tactical Appraisal Prethoryn
				nations.11260 #Tactical Appraisal Unbidden
				nations.11270 #Hire Armies
				nations.11275 #Merc Pays Back Patron
				nations.11280 #Patron Panel
				nations.11285 #Recall Fleet
			}
		}
	}

	immediate = {
		ai_trade_facility = {
			#offer_hire_mercenary_fleet = yes
			#offer_prolong_fleet_contract = yes
			from = {
				#save_event_target_as = mercenary_enclave_country
				#Failsafe to catch if fleets are no longer on loan
				#remove_loaning_fleet_flag = yes
			}
			country_event = { id = nations.11205 }
		}
	}
}
