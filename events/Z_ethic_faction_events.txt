country_event = {
	id = factions.13
	hide_window = yes

			if = {
				limit = { is_war_participant = { who = root side = attackers } }
				root = {
					set_timed_country_flag = {
						flag = entered_war_as_attacker
						years = 20 
					}
				}
			}

				limit = { is_war_participant = { who = root side = defenders } }

					set_timed_country_flag = {
						flag = entered_war_as_defender
						years = 20 
					}
				}
			}
		}
	}
}
