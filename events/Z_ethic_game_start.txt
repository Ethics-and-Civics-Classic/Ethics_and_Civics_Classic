namespace = game_start

			}
			ethic_fanatic_authoritarian = {
				ruler = { change_background_ethic = ethic_authoritarian }
			}
			ethic_fanatic_spiritualist = {
				ruler = { change_background_ethic = ethic_spiritualist }
			}
