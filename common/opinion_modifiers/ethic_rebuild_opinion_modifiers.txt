
triggered_opinion_barbaric_despoilers = {
	trigger = {
		is_country_type = default
		NOR = {
			is_militarist = yes
			is_pacifist = yes
			is_authoritarian = yes
			has_ethic = ethic_gestalt_consciousness
			has_valid_civic = civic_fanatic_purifiers
		}
		NOT = { 
			has_valid_civic = civic_barbaric_despoilers 
		}
		FROM = { 
			has_valid_civic = civic_barbaric_despoilers 
		}
	}
	opinion = {
		base = -50
	}
}
