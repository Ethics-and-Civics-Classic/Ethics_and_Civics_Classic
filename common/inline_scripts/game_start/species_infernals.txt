# Infernals Species Initial Setup
# Rural Districts are set to:
#	* 2× Generator Districts (+1 for Tankbound)
#	* 4× Mining Districts
#	* 0× Agriculture Districts



if = { # if you are a normal infernal and don't need food
	limit = {
		exists = owner
		owner = {
			is_infernal_empire = yes
		}
	}
	set_variable = {
		which = energy_districts
		value = 2
	}
	set_variable = {
		which = mining_districts
		value = 4
	}
	if = {  # If you are any of these, you get an extra food district. 
		limit = {
			owner = {
				OR = {
					has_origin = origin_tree_of_life
					has_origin = origin_here_be_dragons
					has_civic = PrimPlayer_civic_here_be_dragons
				}
			}
		}
		set_variable = {
			which = farming_districts
			value = 1
		}
	}
	else = { #dont need special food or the extra food will be added in game_start/species_subspecies or game_start/civics
		set_variable = {
			which = farming_districts
			value = 0
		}
	}
}

if = {
	limit = {
		exists = owner
		owner = {
			is_tankbound_empire = yes
		}
	}
	change_variable = {
		which = energy_districts
		value = 3
	}
}
