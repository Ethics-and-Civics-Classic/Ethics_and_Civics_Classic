# Machine Cult Inital Setup

# Additional Robotic Assembly only for organics bc individualist machine always start with it
if = {
	limit = {
		exists = owner
		owner = {
			has_civic = civic_machine_cult
		}
	}
	add_building = building_robot_assembly_plant
}

# Mecha Monastery
if = {
	limit = {
		exists = owner
		owner = {
			OR = {
				has_civic = civic_machine_cult
				has_civic = civic_machine_cult_machine
			}
		}
	}
	add_building = building_mecha_monastery
}
if = {
	limit = {
		has_zone = {
			zone = zone_habitat_research_unity
		}
		exists = owner
		owner = {
			has_valid_civic = civic_machine_cult_machine
		}
	}
	add_building = {
		zone = zone_habitat_research_unity
		building = building_organic_sanctuary
	}
}
else_if = {
	limit = {
		exists = owner
		owner = {
			has_valid_civic = civic_machine_cult_machine
		}
	}
	add_building = {
		zone = zone_research_unity
		building = building_organic_sanctuary
	}
}
