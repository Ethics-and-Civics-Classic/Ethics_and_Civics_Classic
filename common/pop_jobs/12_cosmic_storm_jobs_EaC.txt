storm_dancer = {	
	category = specialist
	swappable_data = {
		default = {
			condition_string = SPECIALIST_JOB_TRIGGER
			building_icon = building_storm_summoning_theater
		}
	}
	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}
	possible = {
		entertainer_job_check_trigger = yes
		OR = {
			has_ethic = ethic_green
			owner = { is_individual_machine = yes }
			owner = { is_gestalt = yes }
		}
	}
	resources = {
		category = planet_entertainers 
		produces = {
			unity = 4
			society_research = 1
		}
		produces = {
			trigger = {
				owner = { has_valid_civic = civic_distributism }
			}
			trade = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		pop_ethic_green_attraction_mult = 0.05
	}
	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_add = 800
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}
	weight = {
		weight = @specialist_job_weight
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 0.2
			NOR = { has_citizenship_rights = yes has_trait = trait_mechanical }
		}
		modifier = {
			factor = 0.1
			can_take_servant_job = yes
		}
		modifier = {
			factor = 0.1
			is_enslaved = yes
			has_slavery_type = { type = slavery_indentured }
		}
		modifier = {
			factor = 25
			owner = { is_individual_machine = no }
		}
	}
	inline_script = "jobs/automodding_priority_unity"
}
