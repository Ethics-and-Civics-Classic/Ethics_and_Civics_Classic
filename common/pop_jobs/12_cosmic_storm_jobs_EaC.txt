astrometeorologist = {
	category = specialist
	swappable_data = {
		default = {
			condition_string = SPECIALIST_JOB_TRIGGER
			building_icon = building_astrometeorology_observation_center
			#clothes_texture_index = 3
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_specialist_job
	possible = {
		complex_specialist_job_check_trigger = yes
		OR = {
			has_ethic = ethic_green		#pop_has_ethic - #No Longer Used in Stellaris 4.0 
			OWNER = { is_individual_machine = yes }
		}
	}

	resources = {
		category = planet_astrometeorologists
		produces = {
			physics_research = 6
			engineering_research = 2
			society_research = 2
		}

		produces = {
			trigger = {
				planet = {
					solar_system = {
						OR = {
							has_environmental_effects = yes
							has_star_flag = storm_system
						}
					}
				}
			}
			physics_research = 3
			engineering_research = 1
			society_research = 1
		}
		upkeep = {
			consumer_goods = 2
		}
		upkeep = {
			trigger = {
				planet = {
					has_planet_flag = has_science_ministry
				}
			}
			consumer_goods = 1
		}
	}
	overlord_resources = {
		category = planet_requisitioned_research
		produces = {
			trigger = {
				planet = {
					has_planet_flag = has_science_ministry
				}
			}
			physics_research = 2
			engineering_research = 1
			society_research = 1
		}
		produces = {
			trigger = {
				planet = {
					has_planet_flag = has_science_ministry
					solar_system = {
						OR = {
							has_environmental_effects = yes
							has_star_flag = storm_system
						}
					}
				}
			}
			physics_research = 1
			engineering_research = 1
			society_research = 1
		}
	}
	
	planet_modifier = {
		pop_ethic_green_attraction_mult = 0.05
	}

	weight = {
		weight = @specialist_job_weight
		mult = value:job_weights_research_modifier|JOB|astrometeorologist|
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 2
			has_living_standard = { type = living_standard_academic_privilege }
		}
		modifier = {
			factor = 0.1
			can_take_servant_job = yes
		}
		#modifier = {	#No Longer Used in Stellaris 4.0
		#	factor = 1.5
		#	#has_job = astrometeorologist
		#}
		modifier = {
			factor = 0.8
			is_enslaved = yes
			has_slavery_type = { type = slavery_indentured }
		}
		modifier = {
			factor = 25
			owner = { is_individual_machine = no }
		}
	}
	inline_script = "jobs/automodding_priority_research"
}
