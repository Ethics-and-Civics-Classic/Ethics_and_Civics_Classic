		}
		custom_tooltip = {
			fail_text = "requires_actor_not_machine_terminator"

				limit = { 
					has_origin = origin_broken_shackles

				}
			}
		}
	}

							add_modifier = {
								modifier = culture_shock_diplomacy

						}
					}

							add_modifier = {

						}
					}
