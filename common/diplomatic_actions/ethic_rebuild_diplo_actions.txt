	requires_recipient_independence = no

	action_type = aggressive
	potential = {
		hidden_trigger = {
			FROM = {
				NOT = {
					has_overlord = ROOT
				}
			}
		}
	}
	possible = {
		custom_tooltip = {
			fail_text = "requires_actor_not_fanatic_purifiers"
			OR = {
				is_same_species = from
				NOT = { has_valid_civic = civic_fanatic_purifiers }
			}
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_fanatic_purifiers"
			OR = {
				is_same_species = from
				from = { NOT = { has_valid_civic = civic_fanatic_purifiers } }
			}
		}
		custom_tooltip = {
			fail_text = "requires_actor_not_devouring_swarm"
			NOT = { has_valid_civic = civic_hive_devouring_swarm }
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_devouring_swarm"
			from = { NOT = { has_valid_civic = civic_hive_devouring_swarm } }
		}
		custom_tooltip = {
			fail_text = "requires_actor_not_machine_terminator"
			OR = {
				is_same_species = from
				NOT = { has_civic = civic_machine_terminator }
				AND = {
					has_civic = civic_machine_terminator
					from = {
						OR = {
							has_country_flag = synthetic_empire
							is_machine_empire = yes
							is_individual_machine = yes
						}
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_machine_terminator"
			OR = {
				is_same_species = from
				from = { NOT = { has_civic = civic_machine_terminator } }
				AND = {
					from = { has_civic = civic_machine_terminator }
					OR = {
						has_country_flag = synthetic_empire
						is_machine_empire = yes
						is_individual_machine = yes
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_actor_not_scorched_earth"
			OR = {
				is_genocidal_infernal = no
				from = { is_infernal_empire = yes }
			}
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_scorched_earth"
			OR = {
				is_infernal_empire = yes
				from = { is_genocidal_infernal = no }
			}
		}
		if = {
			limit = {
				from = { has_valid_civic = civic_pompous_purists }
				NOT = { root = { is_overlord_to = from } }
			}
			if = {
				limit = {
					has_intel = {
						who = from
						intel = civics
					}
				}
				custom_tooltip = {
					fail_text = requires_recipient_not_pompous
					always = no
				}
			}
			else = {
				custom_tooltip = {
					fail_text = diplo_action_no_low_intel
					always = no
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_in_galactic_empire"
			NAND = {
				has_galactic_emperor = yes
				is_galactic_community_member = no
				from = { is_galactic_community_member = yes }
			}
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_custodian"
			from = { is_galactic_custodian = no }
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_gal_emperor"
			from = { is_galactic_emperor = no }
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_btc"
			from = { has_been_declared_crisis = no }
		}
		custom_tooltip = {
			fail_text = "requires_not_guaranteeing_independence"
			NOT = {
				is_guaranteeing = from
			}
		}
		if = {
			limit = {
				NOR = {
					has_menace_perk = menp_intimidation
					has_valid_civic = civic_feudal_realm
				}
			}
			custom_tooltip = {
				fail_text = requires_50_trust_or_excellent
				OR = {
					opinion_level = {
						who = from
						level >= excellent
					}
					from = {
						trust = {
							who = root
							value >= 50
						}
					}
					can_ignore_trust_requirements_for_subjects = yes
					AND = {
						has_valid_civic = civic_hive_familiar_face
						from = {
							trust = {
								who = root
								value >= 25
							}
						}
					}
				}
			}
			custom_tooltip = {
				fail_text = requires_embassy_or_diplomatic_networking
				OR = {
					has_embassy = from
					has_active_tradition = tr_diplomacy_diplomatic_networking
					can_ignore_trust_requirements_for_subjects = yes
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_recipient_not_khan"
			from = { NOT = { has_country_flag = heir_of_the_khan_khan } }
		}
	}
	proposable = {
		if = {
			limit = {
				hidden_trigger = { from = { is_country_type = awakened_fallen_empire } }
			}
			hidden_trigger = { always = yes }
		}
	}
	should_ai_propose = {
		modifier = {
			mult = 0
			from = {
				trust = {
					who = root
					value < 50
				}
			}
		}
	}
	ai_acceptance = {
		modifier = {
			add = -100
			from = {
				trust = {
					who = root
					value < 50
				}
				NOT = { has_country_flag = heir_of_the_khan_khan }
			}
			desc = LACK_OF_TRUST
		}
		modifier = {
			add = -50
			root = {
				is_militarist = yes
				has_faction = imperialist
			}
			desc = TOO_IMPERIAL
		}
		modifier = {
			add = 50
			from = {
				has_country_flag = heir_of_the_khan_khan
				is_neighbor_of = root
			}
			desc = NEIGHBORING_KHAN
		}
		modifier = {
			add = 25
			from = {
				has_country_flag = heir_of_the_khan_khan
			}
			any_neighbor_country = {
				is_subject = yes
				any_agreement = { agreement_preset = preset_satrapy }
			}
			desc = NEIGHBORING_SATRAPY
		}
		modifier = {
			add = 50
			from = {
				has_country_flag = heir_of_the_khan_khan
			}
			OR = {
				has_origin = origin_heirs_of_the_khan
				has_origin = origin_heirs_of_the_khan_corporate
			}
			desc = RIGHTFUL_KHAN
		}
		modifier = {
			subtract = value:num_defensive_pacts_times_25
			from = {
				has_country_flag = heir_of_the_khan_khan
			}
			desc = NUMBER_OF_DEFENSIVE_PACTS
		}
		modifier = {
			subtract = 100
			desc = IN_GALACTIC_EMPIRE
			has_galactic_emperor = yes
			is_galactic_community_member = yes
			from = {
				has_country_flag = heir_of_the_khan_khan
			}
		}
		modifier = {
			subtract = 100
			desc = IN_GLACTIC_COMMUNITY_CUSTODIAN
			has_galactic_custodian = yes
			is_galactic_community_member = yes
			from = {
				has_country_flag = heir_of_the_khan_khan
			}
		}
		modifier = {
			subtract = 100
			desc = WANTS_TO_BE_FREE
			is_nomadic = yes
		}
		modifier = {
			subtract = 100
			desc = NOMADIC_OVERLORD
			is_nomadic = no
			from = {
				is_nomadic = yes
			}
		}
		modifier = {
			mult = 0
			from = {
				has_country_flag = heir_of_the_khan_khan
			}
			relative_power = {
				who = from
				category = all
				value >= equivalent
			}
			desc = BLANK_STRING
		}
	}
