	}

			energy = @b1_upkeep
		}
	}
	inline_script = {
		script = jobs/trader_add
		AMOUNT = @building_static_jobs
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				NOT = { has_technology = tech_primitive_holographic_technology }
			}
		}
		modifier = {
			job_trader_add = -100
		}
	}
	triggered_planet_modifier = {
		potential = {

		}
	}

			exists = owner
