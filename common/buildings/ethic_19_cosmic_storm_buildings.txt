building_astrometeorology_observation_center = {
	base_buildtime = @b1_time
	planet_limit = 1
	category = research
	
	building_sets = {
		research
		physics
	}

	potential = {
		exists = owner
		owner = {
			is_astrometeorologist_empire = yes
		}
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				is_astrometeorologist_empire = no
			}
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}

	resources = {
		category = planet_buildings_storm_tech
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			owner = {
				OR = {
					has_ethic = ethic_industrial
					has_ethic = ethic_fanatic_industrial
				}
			}
		}
		modifier = {
			job_astrometeorologist_industrialist_add = 200
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			owner = { 
				OR = {
					has_ethic = ethic_green
					has_ethic = ethic_fanatic_green
				}
			}
		}
		modifier = {
			job_astrometeorologist_add = 200
		}
	}

	# inline_script = {
	#	script = "jobs/astrometeorologists_add"
	#	AMOUNT = 2
	#}

	inline_script = {
		script = cosmic_storms/StormBuildingModifiers
	}
	
	triggered_desc = {
		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_industrial
					has_ethic = ethic_fanatic_industrial
				}
			}
		}
		text = "job_astrometeorologist_industrialist_effect_desc"
	}
	
	triggered_desc = {
		trigger = {
			owner = { 
				OR = {
					has_ethic = ethic_green
					has_ethic = ethic_fanatic_green
				}
			}
		}
		text = "job_astrometeorologist_effect_desc"
	}


	custom_storm_ai_weight = {
		weight = 0
		modifier = {
			add = 10
			solar_system = {
				has_environmental_effects = yes
			}
		}
		modifier = {
			factor = 2
			AND = {
				exists = owner
				solar_system = {
					system_has_storm_attraction_tech = yes
				}
			}
		}
	}
}
