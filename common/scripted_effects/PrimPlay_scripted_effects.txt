PrimPlayer_create_subterranean_nation = {
	optimize_memory
	if = {
		limit = {
			NOT = {
				any_country = {
					is_country_type = PrimPlayer_adversary
					has_country_flag = $FLAG$
				}
			}
		}
		if = {
			limit = { has_toxoids = yes }
			create_species = {
				name = random
				class = TOX
				portrait = random
				traits = {
					trait = trait_cave_dweller
					trait = trait_inorganic_breath
					trait = trait_incubator
					trait = random_traits
				}
				homeworld = root
			}
		}
		else = {
			create_species = {
				name = random
				class = random_non_machine
				portrait = random
				traits = {
					trait = trait_cave_dweller
					trait = random_traits
				}
				homeworld = root
			}
		}
		create_country = {
			name = "NAME_Subterranean_Nation"
			species = last_created_species
			type = PrimPlayer_adversary
			ethos = random
			origin = origin_subterranean
			auto_delete = no
			flag = {
				icon = {
					category = "zoological"
					file = "flag_zoological_10.dds"
				}
				background= {
					category = "backgrounds"
					file = "new_dawn.dds"
				}
				colors={
					"red"
					"black"
					"null"
					"null"
				}
			}
			effect = {
				set_relation_flag = {
					who = root.owner
					flag = subterror
				}
				change_country_flag = {
					icon = {
						category = "pre_ftl"
						file = "preftl_industrial_age.dds"
					}
					background = {
						category = "backgrounds"
						file = "new_dawn.dds"
					}
					colors = {
						"beige"
						"yellow"
						"null"
						"null"
					}
				}
				remove_pre_ftl_flags = yes
				set_pre_ftl_age = industrial_age
				set_country_flag = industrial_age
				add_awareness = 100
				set_country_flag = $FLAG$
				establish_contact = { who = event_target:subterranean_planet_owner@root.owner location = root }
				establish_communications_no_message = event_target:subterranean_planet_owner@root.owner
				save_event_target_as = subterranean_nation@root.owner
				owner_species = { save_event_target_as = subterranean_species@root.owner }
			}
		}
	}
	else_if = {
		limit = {
			NOT = { exists = event_target:subterranean_nation@root.owner }
		}
		random_country = {
			limit = {
				is_country_type = PrimPlayer_adversary
				has_country_flag = $FLAG$
			}
			save_event_target_as = subterranean_nation@root.owner
		}
	}
}
create_scion_lgate_link = {
	capital_scope = {
		solar_system = { save_event_target_as = home_system }
	}
	event_target:scion_master = {
		random_system_within_border = { save_event_target_as = fe_target_system }
		if = { 
			limit = { is_materialist = yes }
			prev = { add_seen_bypass_type = fe_materialist_lgate_base }
		}
		if = { 
			limit = { is_spiritualist = yes }
			prev = { add_seen_bypass_type = fe_spiritualist_lgate_base }
		}
	}
	set_visited = event_target:fe_target_system
	event_target:home_system = {
		event_target:fe_target_system = {
			if = { 
				limit = { event_target:scion_master = { is_materialist = yes } }
				prev = {
					if = {
						limit = { always = yes }
						spawn_megastructure = {
							type = fe_materialist_lgate_base
							orbit_angle = 360
							orbit_distance = 30
						}
						set_star_flag = fe_materialist_lgate

							limit = {
								is_megastructure_type = fe_materialist_lgate_base
								NOT = { has_megastructure_flag = fe_materialist_lgate_activated }
							}
							activate_gateway = this
							set_megastructure_flag = fe_materialist_lgate_activated
						}
					}
				}
				if = {
					limit = { always = yes }
					spawn_megastructure = {
						type = fe_materialist_lgate_base
						orbit_angle = 270
						orbit_distance = 30
					}
					set_star_flag = fe_materialist_lcluster1
					random_system_megastructure = {
						limit = {
							is_megastructure_type = fe_materialist_lgate_base

						random_system_megastructure = {

synth_queen_scion_warning = {

		limit = {
			any_playable_country = {
				OR = {
					has_origin = origin_scion

					has_country_flag = scion_master

			}
		}

			limit = {
				OR = {
					has_origin = origin_scion

					has_civic = PrimPlayer_civic_scion_spiritualist

					has_country_flag = scion_master

			}

		}
	}

		synth_queen_fe_war = yes
	}
